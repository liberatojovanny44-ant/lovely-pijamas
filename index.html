const CACHE = 'lovely-v2'; const BASE = '/lovely-pijamas'; const ASSETS = [ BASE + '/', BASE + '/index.html', BASE + '/manifest.json', 'https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.4.0/css/all.min.css', 'https://cdnjs.cloudflare.com/ajax/libs/xlsx/0.18.5/xlsx.full.min.js' ]; self.addEventListener('install', e => { e.waitUntil( caches.open(CACHE).then(cache => cache.addAll(ASSETS)) ); self.skipWaiting(); }); self.addEventListener('activate', e => { e.waitUntil( caches.keys().then(keys => Promise.all(keys.filter(k => k !== CACHE).map(k => caches.delete(k))) ) ); self.clients.claim(); }); self.addEventListener('fetch', e => { e.respondWith( caches.match(e.request).then(cached => cached || fetch(e.request).catch(() => caches.match(BASE + '/index.html'))) ); });
